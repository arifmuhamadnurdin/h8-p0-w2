2.Tahun Kabisat

procedure tahunKabisat

input tahun

if tahun mod 4 = 0 and tahun mod 100 != 0
	then 'Tahun Kabisat'
else if tahun mod 4 = 0
	then 'Tahun Kabisat'
else if tahun mod 4 = 100
	then 'Tahun Kabisat'
else if tahun mod 4 = 400
	then 'Tahun Kabisat'
else 'Bukan tahun kabisat'
